function startSession() {

  const name  = document.getElementById('setupName').value.trim();

  const role  = document.getElementById('setupRole').value;

  const email = document.getElementById('setupEmail').value.trim();

 

  if (!name || !role) { toast('Preencha nome e cargo'); return; }

 

  // Gerar UID baseado em nome+cargo (estável entre sessões)

  const uid = btoa(encodeURIComponent(name + role)).replace(/[^a-zA-Z0-9]/g,'').substring(0,28);

  sessionUser = { name, role, email, uid };

 

  localStorage.setItem('dd_session', JSON.stringify(sessionUser));

  initApp();

}

 

function initApp() {

  document.getElementById('setupScreen').classList.add('hidden');

  document.getElementById('syncBar').style.display   = 'flex';

  document.getElementById('appLayout').style.display = 'grid';

 

  document.getElementById('syncUser').textContent   = sessionUser.name;

  document.getElementById('sidebarRole').textContent = CARGO_CONFIG.label;

 

  // Pré-preencher email no modal

  if (sessionUser.email) localStorage.setItem('dd_email_dest', sessionUser.email);

 

  // Conectar ao Firebase

  connectFirebase();

  buildNav();

  renderDay(currentDay);

}

 

function connectFirebase() {

  setSyncStatus('syncing', 'Conectando...');

 

  // Listener em tempo real

  try {

    unsubscribe = window._onSnapshot(userDocRef(), (snap) => {

      if (snap.exists()) {

        const data = snap.data();

        state = {

          checkedTasks:  data.checkedTasks  || {},

          completedDays: data.completedDays || {},

          diaryEntries:  data.diaryEntries  || {},

          customTasks:   data.customTasks   || {},

          deletedTasks:  data.deletedTasks  || {},

        };

        renderDay(currentDay);

        buildNav();

        updateProgress();

      }

      setSyncStatus('synced', 'Sincronizado');

    }, (err) => {

      console.error('Firebase error:', err);

      setSyncStatus('offline', 'Offline — dados locais');

    });

  } catch(e) {

    setSyncStatus('offline', 'Firebase não configurado — dados locais');

  }

}

 

function switchUser() {

  if (unsubscribe) unsubscribe();

  localStorage.removeItem('dd_session');

  location.reload();

}

 

// ── Auto-login ──────────────────────────────────────────

window.addEventListener('firebase-ready', () => {

  const saved = localStorage.getItem('dd_session');

  if (saved) {

    try {

      sessionUser = JSON.parse(saved);

      initApp();

    } catch { /* show setup */ }

  }

});

 

// ── Save to Firebase ─────────────────────────────────────

async function persistState() {

  setSyncStatus('syncing', 'Salvando...');

  try {

    await window._setDoc(userDocRef(), {

      ...state,

      lastUpdated: new Date().toISOString(),

      userName:    sessionUser.name,

      userRole:    sessionUser.role,

      userEmail:   sessionUser.email,

      cargo:       CARGO_CONFIG.label,

    }, { merge: true });

    setSyncStatus('synced', 'Salvo ✓');

  } catch(e) {

    // Fallback local

    localStorage.setItem('dd_state_' + sessionUser.uid, JSON.stringify(state));

    setSyncStatus('offline', 'Salvo localmente');

  }

}

}

function buildTaskEl(task) {

  const checked = !!state.checkedTasks[task.id];

  const el = document.createElement('div');

  el.className = `task-item${checked?' checked':''}`;

  el.id = `task-${task.id}`;

  el.setAttribute('onclick',`toggleTask('${task.id}')`);

  el.innerHTML = `

    <div class="task-check"><span class="check-icon">✓</span></div>

    <div class="task-body">

      <div class="task-text">${task.text}</div>

      ${task.meta?`<div class="task-meta">${task.meta}</div>`:''}

    </div>

    <div class="task-actions" onclick="event.stopPropagation()">

      <button class="task-btn" onclick="openEditModal('${task.id}','${esc(task.text)}','${esc(task.meta||'')}')">✏️</button>

      <button class="task-btn del" onclick="deleteTask('${task.id}')">✕</button>

    </div>`;

  return el;

}

 

function esc(s) { return s.replace(/'/g,"\\'").replace(/"/g,'&quot;'); }

 

function toggleTask(id) {

  state.checkedTasks[id] = !state.checkedTasks[id];

  persistState();

  const el = document.getElementById(`task-${id}`);

  if (el) el.classList.toggle('checked', state.checkedTasks[id]);

  updateNavChecks();

  updateProgress();

}

 

function addTask(d) {

  const input = document.getElementById('newTaskInput');

  const text  = input.value.trim();

  if (!text) return;

  const id = `custom-${d}-${Date.now()}`;

  if (!state.customTasks[d]) state.customTasks[d] = [];

  state.customTasks[d].push({id, text});

  persistState();

  input.value = '';

  renderDay(d);

  toast('Tarefa adicionadda ✓');

}

 

function deleteTask(id) {

  state.deletedTasks[id] = true;

  Object.keys(state.customTasks).forEach(d => {

    state.customTasks[d] = (state.customTasks[d]||[]).filter(t=>t.id!==id);

  });

  persistState();

  renderDay(currentDay);

  toast('Tarefa removida');

}

 

function openEditModal(id, text, meta) {

  editingTask = id;

  document.getElementById('modalTaskText').value = text;

  document.getElementById('modalTaskMeta').value = meta;

  document.getElementById('modalOverlay').classList.add('open');

}

function openEmailModal() {

  const saved = localStorage.getItem('dd_email_dest')||'';

  document.getElementById('emailDest').value = saved;

  updateEmailPreview();

  document.getElementById('emailOverlay').classList.add('open');

}

 

function closeEmailModal(e) {

  if (e && e.target!==document.getElementById('emailOverlay')) return;

  document.getElementById('emailOverlay').classList.remove('open');

}

 

function updateEmailPreview() {

  const {subject,body} = buildEmailContent(currentDay);

  document.getElementById('emailSubject').textContent = 'ASSUNTO: '+subject;

  document.getElementById('emailBody').textContent    = body;

}

 

function sendEmail() {

  const dest = document.getElementById('emailDest').value.trim();

  if (!dest||!dest.includes('@')) {

    const bar = document.getElementById('emailStatusBar');

    bar.className='email-status-bar error';

    bar.textContent='⚠️ Informe um e-mail válido.';

    return;

  }

  localStorage.setItem('dd_email_dest', dest);

  const {subject,body} = buildEmailContent(currentDay);

  window.location.href = `mailto:${encodeURIComponent(dest)}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;

 

  state.completedDays[currentDay] = true;

  persistState();

  renderDay(currentDay);

  buildNav();

  updateNavChecks();

 

  const bar = document.getElementById('emailStatusBar');

  bar.className='email-status-bar success';

  bar.textContent='✅ E-mail aberto! Dia marcado como concluíddo. Dados salvos na nuvem.';

  setTimeout(()=>closeEmailModal(), 2200);

  toast('📧 Relatório do Dia '+currentDay+' enviado e salvo na nuvem!');

}

 

function copyEmailContent() {

  const {subject,body} = buildEmailContent(currentDay);

  navigator.clipboard.writeText(`ASSUNTO: ${subject}\n\n${body}`).then(()=>{

    toast('📋 Conteúdo copiado!');

    document.getElementById('emailStatusBar').className='email-status-bar success';

    document.getElementById('emailStatusBar').textContent='✅ Copiado! Cole no seu e-mail.';

  });

}
